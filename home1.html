document.addEventListener('DOMContentLoaded', function() { // 1. Xử lý nút Sở thích (Tag Buttons) const tagButtons = document.querySelectorAll('.tag-button'); tagButtons.forEach(button => { button.addEventListener('click', function() { // Thêm/Xóa class 'selected' để thay đổi kiểu dáng (CSS sẽ định nghĩa) this.classList.toggle('selected'); }); }); // 2. Xử lý nút Đăng ký (Submit Button) const submitButton = document.querySelector('.submit-button'); submitButton.addEventListener('click', function() { // Lấy giá trị từ các trường nhập liệu const ten = document.getElementById('ten').value.trim(); const email = document.getElementById('email').value.trim(); // Lấy danh sách sở thích đã chọn const selectedInterests = Array.from(document.querySelectorAll('.tag-button.selected')) .map(btn => btn.textContent); // Kiểm tra validation đơn giản if (ten === "") { alert("Vui lòng nhập Tên của bạn."); return; } // Tạo thông báo kết quả let message = `Đăng ký thành công!\n\n`; message += `Tên: ${ten}\n`; message += `Email: ${email}\n`; message += `Sở thích đã chọn: ${selectedInterests.join(', ') || 'Không có'}\n\n`; message += `Dữ liệu này sẽ được gửi đi (trên thực tế).`; alert(message); // Ở đây, bạn sẽ thêm logic để gửi dữ liệu đến máy chủ (sử dụng fetch/axios, v.v.) }); });
